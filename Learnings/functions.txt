##Como contar numero de registros(filas)
SELECT COUNT(column_name)
FROM table_name
WHERE condition;

##Como seleccionar solo los registros en una columna dada que sean distintos, (elimina duplicados)
SELECT DISTINCT column1, column2, ...
FROM table_name

##Como redondear decimales
ROUND(f,p) retorna f redondeado a p lugares decimales
SELECT name, ROUND(population/1000000,2), ROUND(gdp/1000000000,2)
FROM world
WHERE continent = 'South America'

##Como filtrar registros cuyas dos atributos tengan igualdad en longitud de caracteres
SELECT name, capital
 FROM world
 WHERE LENGTH(name) LIKE LENGTH(capital)

##Como filtrar los registros cuyas dos atributos coincidan en el primer caracter y sean distintos
SELECT name, capital
FROM world
WHERE LEFT(name, 1) = LEFT(capital, 1) AND name <> capital

##Como filtrar registros que tengan un atributo mayor a un registro especifico
SELECT name FROM world
  WHERE population >
     (SELECT population FROM world
      WHERE name='Russia')

##FILTRAR todos los registros mostrando un atributo en terminos de porcentaje con respecto a un atributo de un registro especifico
SELECT name, CONCAT(ROUND((population*100)/(SELECT population FROM world WHERE name = "Germany"),0),"%")
FROM world
WHERE continent = "Europe"

##SUBQUERYS correlacionadas (Como filtrar el registro alfabeticamnete mas pequeÃ±o por cada grupo segun atributo, en este caso filtrar el primer pais(alfabeticamente) de cada continente)
SELECT continent, name
FROM world w1
WHERE name = (SELECT MIN(name) FROM world w2 WHERE w2.continent = w1.continent)

##Como sumar la poblacion de 3 paises
SELECT SUM(population)
FROM world
WHERE name in ("Estonia","Latvia","Lithuania")

##Como filtrar grupos
The HAVING clause was added to SQL because the WHERE keyword cannot be used with aggregate functions.
SELECT continent, SUM(population)
  FROM world
 GROUP BY continent
HAVING SUM(population)>500000000
ORDER BY SUM(population) DESC

(INNER) JOIN: Returns records that have matching values in both tables
LEFT (OUTER) JOIN: Returns all records from the left table, and the matched records from the right table
RIGHT (OUTER) JOIN: Returns all records from the right table, and the matched records from the left table
FULL (OUTER) JOIN: returns all matching records from both tables whether the other table matches or not.

Syntaxis:
SELECT column_name(s)
FROM table1
(INNER|LEFT|RIGHT|FULL) JOIN table2
ON table1.column_name = table2.column_name;

The UNION operator is used to combine the result-set of two or more SELECT statements.
SELECT column_name(s) FROM table1
UNION
SELECT column_name(s) FROM table2;
