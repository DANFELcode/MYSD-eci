##Como contar numero de registros(filas) en SQL
SELECT COUNT(column_name)
FROM table_name
WHERE condition;

##Como seleccionar solo los registros en una columna dada que sean distintos, (elimina duplicados)
SELECT DISTINCT column1, column2, ...
FROM table_name

##Como redondear decimales
ROUND(f,p) retorna f redondeado a p lugares decimales
SELECT name, ROUND(population/1000000,2), ROUND(gdp/1000000000,2)
FROM world
WHERE continent = 'South America'

##Como filtrar registros cuyas dos atributos tengan igualdad en longitud de caracteres
SELECT name, capital
 FROM world
 WHERE LENGTH(name) LIKE LENGTH(capital)

##Como filtrar los registros cuyas dos atributos coincidan en el primer caracter y sean distintos
SELECT name, capital
FROM world
WHERE LEFT(name, 1) = LEFT(capital, 1) AND name <> capital
